names_all = []

for year in range(1880,2014+1): #the list is extended by 1 to include 2014
    names_all.append(pd.read_csv('names/yob{}.txt'.format(year),names=['name','sex','number']))
    #we have appended to the list, the result of reading a CSV file
    #the number will be replaced in the generic string {}
    #Also, we are explicitly mentioning column names
    names_all[-1]['year'] = year
    #we are adding a column for the year at the end, hence -1

allyears = pd.concat(names_all)
